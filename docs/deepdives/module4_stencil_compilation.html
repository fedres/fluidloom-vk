2. Generated GLSL

#version 460
#extension GL_EXT_buffer_reference : require

layout(buffer_reference, scalar) buffer density_Buffer { float data[]; };
layout(buffer_reference, scalar) buffer velocity_Buffer { vec3 data[]; };

layout(push_constant) uniform FieldTable {
    uint64_t density;
    uint64_t velocity;
    uint64_t density_new;
    uint32_t activeVoxelCount;
} fields;

#define Read_density(idx) (density_Buffer(fields.density).data[idx])
#define Read_velocity(idx) (velocity_Buffer(fields.velocity).data[idx])
#define Write_density_new(idx, val) (density_Buffer(fields.density_new).data[idx] = val)

layout(local_size_x = 128) in;

void main() {
    uint idx = gl_GlobalInvocationID.x;
    if (idx >= fields.activeVoxelCount) return;
    
    // --- User Code ---
    vec3 vel = Read_velocity(idx);
    float d = Read_density(idx);
    // ... advection logic ...
    Write_density_new(idx, d);
}

3. Compilation Process

std::vector compileToSPIRV(const string& glsl) {
    // Write GLSL to temp file
    string glslFile = "/tmp/shader_" + randomID() + ".comp";
    ofstream(glslFile) << glsl;
    
    // Invoke glslc
    string cmd = "glslc -fshader-stage=compute -o " + spvFile + " " + glslFile;
    int ret = system(cmd.c_str());
    
    if (ret != 0) throw runtime_error("Compilation failed");
    
    // Read SPIR-V
    ifstream in(spvFile, ios::binary);
    vector spirv(fileSize / 4);
    in.read((char*)spirv.data(), fileSize);
    
    // Cleanup temp files
    remove(glslFile.c_str());
    remove(spvFile.c_str());
    
    return spirv;
}

4. Pipeline Creation

vk::Pipeline createComputePipeline(const vector& spirv) {
    // Create shader module
    vk::ShaderModuleCreateInfo moduleInfo{
        .codeSize = spirv.size() * sizeof(uint32_t),
        .pCode = spirv.data()
    };
    vk::ShaderModule module = device.createShaderModule(moduleInfo);
    
    // Create pipeline (with cache)
    vk::ComputePipelineCreateInfo pipelineInfo{
        .stage = {.module = module, .pName = "main"},
        .layout = m_pipelineLayout
    };
    
    auto result = device.createComputePipeline(m_pipelineCache, pipelineInfo);
    
    device.destroyShaderModule(module);  // Cleanup
    return result.value;
}
