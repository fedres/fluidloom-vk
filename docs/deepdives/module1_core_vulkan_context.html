2. VulkanContext Initialization

The initialization process follows a strict sequence to ensure proper Vulkan setup:

void VulkanContext::init(bool enableValidation) {
    // 1. Initialize volk meta-loader
    volkInitialize();
    
    // 2. Setup Vulkan-HPP dispatcher
    VULKAN_HPP_DEFAULT_DISPATCHER.init(vkGetInstanceProcAddr);
    
    // 3. Create instance with vk-bootstrap
    auto instance = vkb::InstanceBuilder()
        .set_app_name("FluidEngine")
        .require_api_version(1, 3, 0)
        .build();
    
    // 4. Select physical device (prefer discrete GPU)
    auto physDev = vkb::PhysicalDeviceSelector(instance)
        .prefer_gpu_device_type(vkb::PreferredDeviceType::discrete)
        .select();
    
    // 5. Create logical device with required features
    vk::PhysicalDeviceVulkan12Features features12{
        .bufferDeviceAddress = VK_TRUE,
        .timelineSemaphore = VK_TRUE,
        .descriptorIndexing = VK_TRUE
    };
    
    vk::PhysicalDeviceVulkan13Features features13{
        .synchronization2 = VK_TRUE,
        .pNext = &features12
    };
    
    // 6. Get queue handles
    m_queues.compute = device.get_queue(vkb::QueueType::compute);
    m_queues.transfer = device.get_queue(vkb::QueueType::transfer);
}

3. Memory Allocation

VMA integration provides efficient memory management with C++ wrapper:

Buffer MemoryAllocator::createBuffer(vk::DeviceSize size, 
                                      vk::BufferUsageFlags usage) {
    // Create buffer using C++ API
    vk::BufferCreateInfo bufferInfo{
        .size = size,
        .usage = usage | vk::BufferUsageFlagBits::eShaderDeviceAddress,
        .sharingMode = vk::SharingMode::eExclusive
    };
    
    // Allocate with VMA
    VkBuffer rawBuffer;
    vmaCreateBuffer(m_allocator, &bufferInfo, &allocInfo, 
                    &rawBuffer, &buffer.allocation, nullptr);
    
    // Wrap in C++ type and get device address
    buffer.handle = vk::Buffer(rawBuffer);
    buffer.deviceAddress = m_device.getBufferAddress({.buffer = buffer.handle});
    
    return buffer;
}

4. GPU Upload Pipeline

Efficient CPUâ†’GPU data transfer using staging buffers:

void MemoryAllocator::uploadToGPU(const Buffer& dst, const void* data, 
                                   vk::DeviceSize size) {
    // 1. Create staging buffer (host-visible)
    Buffer staging = createBuffer(size, 
        vk::BufferUsageFlagBits::eTransferSrc,
        VMA_MEMORY_USAGE_CPU_ONLY);
    
    // 2. Copy data to staging
    memcpy(staging.mappedData, data, size);
    vmaFlushAllocation(m_allocator, staging.allocation, 0, size);
    
    // 3. Submit transfer command
    vk::CommandBuffer cmd = m_context.beginSingleTimeCommands(pool);
    cmd.copyBuffer(staging.handle, dst.handle, {{0, 0, size}});
    m_context.endSingleTimeCommands(cmd, pool, queue);
    
    // 4. Cleanup staging buffer
    destroyBuffer(staging);
}

ğŸ” Best Practices

Type Safety

All code uses vk:: namespace (C++ bindings) instead of raw C API. This provides compile-time type checking and RAII resource management: 

// âœ… Correct: C++ bindings
vk::CommandBuffer cmd;
vk::BufferCreateInfo bufferInfo{...};

// âŒ Avoid: C API
VkCommandBuffer cmd;
VkBufferCreateInfo bufferInfo = {...};

Error Handling

C++ bindings throw vk::SystemError exceptions by default. All initialization code is wrapped in try-catch blocks with LOG_CHECK for assertions: 

try {
    auto result = instanceBuilder.build();
    LOG_CHECK(result.has_value(), "Instance creation failed");
} catch (const vk::SystemError& e) {
    LOG_ERROR("Vulkan error: {}", e.what());
    throw;
}

Resource Cleanup

RAII ensures automatic cleanup, but explicit order is maintained in destructors: 

~VulkanContext() {
    if (m_device) {
        m_device.waitIdle();  // Wait for GPU completion
        m_device = nullptr;    // Automatic cleanup via RAII
    }
    if (m_instance) {
        m_instance = nullptr;  // Instance destroyed last
    }
}
