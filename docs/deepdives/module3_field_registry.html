BDA Table (Device Address Array)

  • Size: 256 entries Ã— 8 bytes = 2KB
  • Memory: CPU_TO_GPU (host-mapped, persistent)
  • Usage: Stores device addresses for each field buffer
  • Access: Via push constants in shaders

Field Buffers

  • One buffer per field (SoA layout)
  • Size: activeVoxelCount Ã— elementSize
  • Usage: STORAGE_BUFFER | SHADER_DEVICE_ADDRESS
  • Initialization: Zero-fill or compute shader splat

1. Field Registration

const FieldDesc& registerField(const string& name, vk::Format format) {
    // Check duplicates and limits
    if (m_fields.count(name) > 0) throw runtime_error("Duplicate field");
    if (m_nextDescriptorIndex >= MAX_FIELDS) throw runtime_error("MAX_FIELDS exceeded");
    
    // Calculate element size from format
    uint32_t elementSize = getElementSize(format);  // 4, 8, 12, or 16 bytes
    
    // Allocate GPU buffer
    vk::DeviceSize bufferSize = m_activeVoxelCount * elementSize;
    auto buffer = m_allocator.createBuffer(
        bufferSize,
        vk::BufferUsageFlagBits::eStorageBuffer |
        vk::BufferUsageFlagBits::eShaderDeviceAddress |
        vk::BufferUsageFlagBits::eTransferDst
    );
    
    // Update BDA table
    uint32_t idx = m_nextDescriptorIndex++;
    m_bdaTableMapped[idx] = static_cast(buffer.deviceAddress);
    
    // Store descriptor
    m_fields[name] = FieldDesc{name, format, elementSize, buffer, idx, buffer.deviceAddress};
    
    return m_fields[name];
}

2. GLSL Code Generation

string generateGLSLHeader() const {
    stringstream ss;
    
    // Extensions
    ss << "#version 460\n";
    ss << "#extension GL_EXT_buffer_reference : require\n";
    ss << "#extension GL_EXT_buffer_reference_uvec2 : require\n\n";
    
    // Buffer reference types
    for (auto& [name, desc] : m_fields) {
        ss << "layout(buffer_reference, scalar) buffer " << name << "_Buffer { "
           << desc.getGLSLType() << " data[]; };\n";
    }
    
    // Push constant field table
    ss << "\nlayout(push_constant) uniform FieldTable {\n";
    for (auto& [name, desc] : m_fields) {
        ss << "    uint64_t " << name << ";\n";
    }
    ss << "} fields;\n\n";
    
    // Accessor macros
    for (auto& [name, desc] : m_fields) {
        ss << "#define Read_" << name << "(idx) ("
           << name << "_Buffer(fields." << name << ").data[idx])\n";
        ss << "#define Write_" << name << "(idx, val) ("
           << name << "_Buffer(fields." << name << ").data[idx] = val)\n";
    }
    
    return ss.str();
}

3. Field Initialization

// Strategy 1: Zero-fill (fastest, no shader compilation)
cmd.fillBuffer(fieldBuffer.handle, 0, bufferSize, 0);

// Strategy 2: Splat with compute shader (for non-zero values)
void initializeField(const string& name, const void* value) {
    if (!m_fillPipeline) {
        createFillPipeline();  // Lazy compilation
    }
    
    const auto& desc = getField(name);
    float val = *static_cast(value);
    
    vk::CommandBuffer cmd = beginSingleTimeCommands(pool);
    cmd.bindPipeline(vk::PipelineBindPoint::eCompute, m_fillPipeline);
    
    struct PC {
        uint64_t bufferAddr;
        uint32_t elementCount;
        float value;
    } pc = {desc.deviceAddress, m_activeVoxelCount, val};
    
    cmd.pushConstants(m_fillLayout, vk::ShaderStageFlagBits::eCompute, 0, pc);
    cmd.dispatch((m_activeVoxelCount + 127) / 128, 1, 1);
    
    endSingleTimeCommands(cmd, pool, queue);
}

4. Shader Usage Example

// Generated GLSL header is included in all compute shaders
#include "generated/fields.glsl"

// Stencil compute shader
layout(local_size_x = 128) in;

void main() {
    uint idx = gl_GlobalInvocationID.x;
    if (idx >= activeVoxelCount) return;
    
    // Use generated accessor macros
    float d = Read_density(idx);
    vec3 v = Read_velocity(idx);
    
    // Compute advection
    vec3 newVel = advect(v, d);
    
    // Write back
    Write_velocity(idx, newVel);
}

Why BDA over Descriptor Sets?

Traditional Vulkan uses descriptor sets for buffer binding, which requires:

  • Descriptor set layout creation
  • Descriptor pool allocation
  • Descriptor set allocation per frame
  • Binding via vkCmdBindDescriptorSets

Buffer Device Address (BDA) eliminates all of this:

  • âœ… Direct 64-bit pointer to GPU memory
  • âœ… Passed via push constants (instant, no allocation)
  • âœ… Dynamic field registration without layout changes
