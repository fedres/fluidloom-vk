1. Buffer Allocation (HaloManager)

HaloBufferSet allocateHalos(FieldDesc& field, SubDomain& domain) {
    HaloBufferSet set;
    uint32_t thickness = 2;  // 2 voxels for 2nd-order stencils
    
    // Calculate face dimensions
    auto dim = domain.bounds.dim();
    std::array faceSizes = {
        thickness * dim.y * dim.z,  // -X
        thickness * dim.y * dim.z,  // +X
        dim.x * thickness * dim.z,  // -Y
        dim.x * thickness * dim.z,  // +Y
        dim.x * dim.y * thickness,  // -Z
        dim.x * dim.y * thickness   // +Z
    };
    
    // Allocate buffers for each face
    for (int face = 0; face < 6; face++) {
        vk::DeviceSize size = faceSizes[face] * field.elementSize;
        
        set.localHalos[face] = allocator.createBuffer(
            size,
            vk::BufferUsageFlagBits::eStorageBuffer |
            vk::BufferUsageFlagBits::eTransferDst);
        
        set.remoteHalos[face] = allocator.createBuffer(
            size,
            vk::BufferUsageFlagBits::eStorageBuffer |
            vk::BufferUsageFlagBits::eTransferSrc);
    }
    
    return set;
}

2. Pack Shader (HaloSync)

// pack.comp - Extract boundary voxels
#version 460
#extension GL_EXT_buffer_reference : require
#extension GL_EXT_scalar_block_layout : require

layout(local_size_x = 128) in;

layout(push_constant) uniform PC {
    uint64_t fieldAddr;
    uint64_t haloAddr;
    uint32_t offset;
    uint32_t count;
} pc;

layout(buffer_reference, scalar) buffer FieldBuf { float data[]; };
layout(buffer_reference, scalar) buffer HaloBuf { float data[]; };

void main() {
    uint idx = gl_GlobalInvocationID.x;
    if (idx >= pc.count) return;
    
    // Copy from field to halo
    float val = FieldBuf(pc.fieldAddr).data[pc.offset + idx];
    HaloBuf(pc.haloAddr).data[idx] = val;
}

3. Halo Exchange (GraphExecutor)

void recordHaloExchange(vk::CommandBuffer cmd, SubDomain& domain) {
    // 1. PACK: Extract boundary data
    for (auto& [fieldName, field] : fields) {
        auto& halos = haloManager.getHaloBufferSet(fieldName, domain.gpuIndex);
        for (auto& neighbor : domain.neighbors) {
            haloSync.recordHaloPack(cmd, field.buffer, halos.remoteHalos[neighbor.face], ...);
        }
    }
    
    // 2. BARRIER: Compute â†’ Transfer
    vk::MemoryBarrier barrier{
        .srcAccessMask = vk::AccessFlagBits::eShaderWrite,
        .dstAccessMask = vk::AccessFlagBits::eTransferRead
    };
    cmd.pipelineBarrier(vk::PipelineStageFlagBits::eComputeShader,
                        vk::PipelineStageFlagBits::eTransfer,
                        vk::DependencyFlags{}, barrier, nullptr, nullptr);
    
    // 3. TRANSFER: Copy to neighbor's local halo
    for (auto& neighbor : domain.neighbors) {
        uint32_t oppositeFace = (neighbor.face % 2 == 0) ? neighbor.face + 1 : neighbor.face - 1;
        auto& neighborHalos = haloManager.getHaloBufferSet(fieldName, neighbor.gpuIndex);
        
        haloSync.recordHaloTransfer(cmd,
            halos.remoteHalos[neighbor.face],
            neighborHalos.localHalos[oppositeFace],
            size);
    }
    
    // 4. BARRIER: Transfer â†’ Compute
    barrier = {
        .srcAccessMask = vk::AccessFlagBits::eTransferWrite,
        .dstAccessMask = vk::AccessFlagBits::eShaderRead
    };
    cmd.pipelineBarrier(vk::PipelineStageFlagBits::eTransfer,
                        vk::PipelineStageFlagBits::eComputeShader,
                        vk::DependencyFlags{}, barrier, nullptr, nullptr);
    
    // 5. UNPACK: Write to field buffer
    for (auto& neighbor : domain.neighbors) {
        haloSync.recordHaloUnpack(cmd, halos.localHalos[neighbor.face], field.buffer, ...);
    }
}
